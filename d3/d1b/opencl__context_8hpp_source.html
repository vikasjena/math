189  inline void compile_kernel_group(const char* kernel_name) {
190  char temp[100];
191  int local = 32;
192  int gpu_local_max = std::sqrt(max_workgroup_size());
193  if (gpu_local_max < local)
194  local = gpu_local_max;
195  snprintf(temp, sizeof(temp), "-D TS=%d -D TS1=%d -D TS2=%d ", local, local,
196  local);
197  std::string kernel_source = "";
198  const char* kernel_group = kernel_info()[kernel_name].group;
199  for (auto kern : kernel_info()) {
200  if (strcmp(kern.second.group, kernel_group) == 0) {
201  kernel_source += kern.second.raw_code;
202  }
203  }
204 
205  try {
206  cl::Program::Sources source(
207  1,
208  std::make_pair(kernel_source.c_str(), strlen(kernel_source.c_str())));
209  cl::Program program_ = cl::Program(context(), source);
210  program_.build({device()}, temp);
211 
212  cl_int err = CL_SUCCESS;
213  // Iterate over the kernel list and get all the kernels from this group
214  // and mark them as compiled.
215  for (auto kern : kernel_info()) {
216  if (strcmp(kern.second.group, kernel_group) == 0) {
217  opencl_context_base::getInstance().kernels[(kern.first)]
218  = cl::Kernel(program_, kern.first, &err);
219  kern.second.exists = true;
220  }
221  }
222  } catch (const cl::Error& e) {
223  check_opencl_error("Kernel Compilation", e);
224  }
225  }
